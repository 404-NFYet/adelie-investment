Q1 JWT 토큰 관리 전체 흐름은? (access 60min + refresh 7일, HttpOnly 쿠키) +

Adelie Investment는 Access Token + Refresh Token 이중 구조를 사용합니다. 두 토큰 모두 HttpOnly 쿠키로 전달되어 JavaScript에서 직접 접근할 수 없으며, XSS 공격으로부터 보호됩니다.

토큰 생성 흐름

fastapi/app/services/auth_service.py 
def _build_token(subject: str, expires_in: int, extra_claims: dict | None = None) -> str:
    settings = get_settings()
    now = datetime.now(timezone.utc)
    payload = {
        "sub": subject,        # 사용자 이메일
        "iat": int(now.timestamp()),
        "exp": int((now + timedelta(seconds=expires_in)).timestamp()),
    }
    if extra_claims:
        payload.update(extra_claims)
    return jwt.encode(
        payload,
        _get_jwt_key(settings.JWT_SECRET),
        algorithm=settings.JWT_ALGORITHM,  # HS256
    )

  2. API 요청 → get_current_user가 쿠키 또는 Authorization 헤더에서 토큰 추출 → JWT 디코딩 → DB 사용자 조회

Q2 Refresh Token 재사용 탐지(reuse detection)는? (JTI 해시 비교, 전체 revoke) +

Refresh Token Rotation + Reuse Detection 패턴을 구현하여, 도난된 토큰의 재사용을 탐지하고 해당 사용자의 모든 리프레시 토큰을 즉시 폐기합니다.

저장 구조 (Redis)

  • auth:rt:{jti} → SHA-256 해시값 (토큰별 키, TTL 7일)
  • auth:rt:uid:{user_id} → SET of JTIs (사용자별 토큰 목록)
fastapi/app/services/auth_service.py 
def _hash_token(token: str) -> str:
    return hashlib.sha256(token.encode("utf-8")).hexdigest()

async def _store_refresh_token(*, user_id: int, jti: str, token_hash: str, ttl: int) -> None:
    client = await _require_redis()
    token_key = f"auth:rt:{jti}"
    user_key = f"auth:rt:uid:{user_id}"
    async with client.pipeline(transaction=True) as pipe:
        pipe.setex(token_key, ttl, token_hash)   # JTI별 해시 저장
        pipe.sadd(user_key, jti)                  # 사용자 JTI 목록에 추가
        pipe.expire(user_key, ttl)
        await pipe.execute()

재사용 탐지 로직

리프레시 요청 시 JTI로 저장된 해시와 들어온 토큰의 해시를 비교합니다. 불일치 또는 부재 시 토큰 재사용으로 판단하여 해당 사용자의 모든 리프레시 토큰을 즉시 폐기(revoke)합니다.

async def refresh_tokens(db: AsyncSession, *, refresh_token: str) -> dict:
    # 1. JWT 디코딩 → jti, uid, email 추출
    payload = jwt.decode(refresh_token, ...)
    jti = payload.get("jti")

    # 2. Redis에서 저장된 해시와 비교
    stored_hash = await client.get(f"auth:rt:{jti}")
    incoming_hash = _hash_token(refresh_token)

    if not stored_hash or stored_hash != incoming_hash:
        # 재사용 감지! 해당 사용자 모든 RT 폐기
        await _revoke_user_refresh_tokens(user.id)
        raise HTTPException(status_code=401, detail="Refresh token reuse detected")

    # 3. 기존 토큰 삭제 + 새 토큰 발급 (Rotation)
    async with client.pipeline(transaction=True) as pipe:
        pipe.delete(f"auth:rt:{jti}")              # 기존 JTI 삭제
        pipe.srem(user_key, jti)                   # 목록에서 제거
        pipe.setex(f"auth:rt:{new_jti}", ttl, _hash_token(new_token))  # 새 JTI 저장
        pipe.sadd(user_key, new_jti)               # 목록에 추가
        await pipe.execute()

전체 폐기 (Revoke All)

async def _revoke_user_refresh_tokens(user_id: int) -> None:
    client = await _require_redis()
    user_key = f"auth:rt:uid:{user_id}"
    jtis = await client.smembers(user_key)         # 사용자의 모든 JTI 조회
    async with client.pipeline(transaction=True) as pipe:
        for jti in jtis:
            pipe.delete(f"auth:rt:{jti}")          # 개별 토큰 키 삭제
        pipe.delete(user_key)                      # 사용자 JTI 목록 삭제
        await pipe.execute()

이 구조의 핵심은 한 번 사용된 리프레시 토큰은 즉시 무효화되고 새 토큰으로 교체(Rotation)된다는 점입니다. 공격자가 탈취한 토큰을 재사용하면 해시 불일치가 발생하여 탐지되고, 정상 사용자의 토큰까지 모두 폐기하여 재로그인을 강제합니다.

Q3 CSRF 방어 구현 상세는? (double-submit cookie, exempt 엔드포인트 4개) +

Double-Submit Cookie 패턴을 사용합니다. 로그인/회원가입/리프레시 시 서버가 CSRF 토큰을 non-HttpOnly 쿠키로 설정하고, 이후 상태 변경 요청(POST/PUT/DELETE) 시 클라이언트가 해당 값을 X-CSRF-Token 헤더로 함께 전송해야 합니다.

CSRF 토큰 발급

fastapi/app/api/routes/auth.py 
def _set_csrf_cookie(response: Response) -> str:
    settings = get_settings()
    token = secrets.token_urlsafe(32)     # 암호학적으로 안전한 랜덤 토큰
    response.set_cookie(
        key=settings.AUTH_CSRF_COOKIE_NAME,   # "csrfToken"
        value=token,
        httponly=False,    # JS에서 읽을 수 있어야 헤더로 보낼 수 있음
        secure=settings.AUTH_CSRF_COOKIE_SECURE,
        samesite=settings.AUTH_CSRF_COOKIE_SAMESITE,  # "lax"
        path=settings.AUTH_CSRF_COOKIE_PATH,
        domain=settings.AUTH_CSRF_COOKIE_DOMAIN,
    )
    return token

CSRF 검증 미들웨어

fastapi/app/main.py 
@app.middleware("http")
async def csrf_middleware(request: Request, call_next):
    settings = get_settings()
    # GET, HEAD, OPTIONS는 상태 변경이 없으므로 면제
    if request.method in {"GET", "HEAD", "OPTIONS"}:
        return await call_next(request)

    path = request.url.path
    # 4개 인증 엔드포인트는 CSRF 면제 (토큰 발급 전이므로)
    csrf_exempt_prefixes = {
        "/api/v1/auth/login",
        "/api/v1/auth/register",
        "/api/v1/auth/refresh",
        "/api/v1/auth/csrf",
    }
    if path in csrf_exempt_prefixes:
        return await call_next(request)

    # CSRF 쿠키가 없으면 = Authorization 헤더 인증 (CSRF 불필요)
    csrf_cookie = request.cookies.get(settings.AUTH_CSRF_COOKIE_NAME)
    if not csrf_cookie:
        return await call_next(request)

    # Double-Submit: 쿠키값 == 헤더값 비교
    csrf_header = request.headers.get(settings.AUTH_CSRF_HEADER_NAME)  # "X-CSRF-Token"
    if not csrf_header or csrf_cookie != csrf_header:
        return JSONResponse(
            status_code=403,
            content={"detail": "CSRF token missing or invalid"},
        )
    return await call_next(request)

Q4 비밀번호 해싱은? (bcrypt, CryptContext, auto deprecated) +

passlib의 CryptContext를 사용하여 bcrypt 해싱을 수행합니다. deprecated="auto" 설정으로 향후 더 강력한 해싱 알고리즘이 추가되면 기존 bcrypt 해시를 자동으로 마이그레이션합니다.

동작 원리

기능설명
schemes=["bcrypt"]bcrypt를 기본 해싱 알고리즘으로 사용
deprecated="auto"현재 기본 스키마 외의 해시를 자동으로 "deprecated" 마킹
_pwd_context.hash()랜덤 salt 생성 + bcrypt 해싱 (cost factor 기본값 12)
_pwd_context.verify()입력 비밀번호와 저장된 해시 비교, deprecated 해시면 True 반환 + needs_update 플래그

bcrypt는 의도적으로 느린 해싱 알고리즘(key stretching)을 사용하여 무차별 대입 공격에 대한 비용을 극대화합니다. salt가 해시 자체에 포함되어 있어 별도 salt 컬럼이 필요 없습니다.

Q5 Rate Limiting Lua 스크립트 동작 원리는? (ZSET, ZREMRANGEBYSCORE, ZCARD) +

Redis Sorted Set(ZSET) 기반 슬라이딩 윈도우 알고리즘을 Lua 스크립트로 원자적(atomic)으로 실행합니다. IP당 100 req/min 제한이며, Redis 장애 시 graceful fallback(허용)으로 가용성을 보장합니다.

Lua 스크립트 전문

fastapi/app/middleware/rate_limit.py 
SLIDING_WINDOW_LUA = """
local key = KEYS[1]                          -- rate limit key (IP별)
local now = tonumber(ARGV[1])                -- 현재 시각 (ms)
local window = tonumber(ARGV[2])             -- 윈도우 크기 (60000ms)
local limit = tonumber(ARGV[3])              -- 최대 허용 수 (100)

-- 1단계: 윈도우 밖의 오래된 요청 제거
redis.call('ZREMRANGEBYSCORE', key, 0, now - window)

-- 2단계: 현재 윈도우 내 요청 수 조회
local count = redis.call('ZCARD', key)

if count < limit then
    -- 3단계(허용): 현재 시각을 score+member로 추가
    redis.call('ZADD', key, now, now)
    redis.call('EXPIRE', key, math.ceil(window / 1000))  -- TTL 설정
    return {1, limit - count - 1, 0}         -- {허용, 남은수, 0}
else
    -- 3단계(차단): 가장 오래된 요청으로 reset 시각 계산
    local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
    local reset = oldest[2] and (tonumber(oldest[2]) + window) or (now + window)
    return {0, 0, reset}                     -- {차단, 0, reset_ms}
end
"""

미들웨어 적용

class RateLimitMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        if request.url.path in _EXEMPT_PATHS:   # /metrics, /docs 등 제외
            return await call_next(request)

        client_ip = request.client.host if request.client else "unknown"
        try:
            cache = await get_redis_cache()
            if cache.client:
                now_ms = int(time.time() * 1000)
                k = key_rate_limit("ip", client_ip)  # "dev:rl:ip:1.2.3.4"
                result = await cache.client.eval(
                    SLIDING_WINDOW_LUA, 1, k, now_ms, WINDOW_MS, RATE_LIMIT
                )
                allowed, remaining, reset_ms = result
                if not allowed:
                    retry_after = max(1, int((reset_ms - now_ms) / 1000))
                    return Response(
                        status_code=429,
                        headers={
                            "X-RateLimit-Limit": str(RATE_LIMIT),
                            "X-RateLimit-Remaining": "0",
                            "X-RateLimit-Reset": str(int(reset_ms / 1000)),
                            "Retry-After": str(retry_after),
                        },
                    )
        except Exception as e:
            logger.warning(f"RateLimit Redis error (fallback=allow): {e}")

동작 시퀀스

  1. ZREMRANGEBYSCORE: score가 0 ~ (now - 60000)인 멤버 제거 → 1분 이전 기록 정리
  2. ZCARD: 남은 멤버 수 = 최근 1분간 요청 수
  3. 요청 수 < 100이면 ZADD로 현재 타임스탬프 추가, 아니면 429 반환

Redis 키 네임스페이스

def key_rate_limit(scope: str, identifier: str) -> str:
    return f"{ENV}:rl:{scope}:{identifier}"   # 예: "dev:rl:ip:192.168.1.1"

Lua 스크립트는 Redis 서버에서 원자적으로 실행되므로, 다중 프로세스/스레드 환경에서도 race condition 없이 정확한 카운팅이 보장됩니다.

def registration_blocked_domains(self) -> List[str]:
    """Parse blocked email domains into a list."""

def _validate_email_domain(email: str, blocked_domains: list[str]) -> str:

    existing = await db.execute(select(User).where(User.email == email))
    if existing.scalar_one_or_none():
        raise HTTPException(status_code=400, detail="이미 등록된 이메일입니다.")

    # 사용자명 미입력 시 이메일 로컬파트 사용
    if not username:
        username = email.split("@", 1)[0]

    # 금칙어 패턴 검증 (re.IGNORECASE)
    pattern = re.compile(settings.REGISTRATION_BLOCKED_USERNAME_PATTERN, re.IGNORECASE)
    _validate_username(username, pattern)

    # 사용자명 중복 체크
    existing_username = await db.execute(select(User).where(User.username == username))
    if existing_username.scalar_one_or_none():
        raise HTTPException(status_code=400, detail="이미 사용 중인 사용자명입니다.")

Q8 SQL Injection 방어는? (SQLAlchemy ORM, parameterized queries) +

Adelie Investment는 SQLAlchemy ORM을 전면 사용하여 모든 데이터베이스 쿼리가 파라미터화된 쿼리(Parameterized Queries)로 실행됩니다. 원시 SQL 문자열 조합이 없으므로 SQL Injection 공격이 구조적으로 불가능합니다.

ORM 쿼리 예시 (사용자 조회)

fastapi/app/services/auth_service.py 
# 이메일로 사용자 조회 — ORM이 자동으로 파라미터 바인딩
result = await db.execute(select(User).where(User.email == email))
user = result.scalar_one_or_none()

위 코드는 SQLAlchemy가 내부적으로 다음과 같은 파라미터화된 SQL로 변환합니다:

-- SQLAlchemy가 생성하는 실제 SQL (파라미터 바인딩)
SELECT users.id, users.email, users.password_hash, ...
FROM users
WHERE users.email = $1
-- $1 = 'user@example.com' (파라미터로 전달, 문자열 조합 아님)

다른 쿼리 패턴들

fastapi/app/core/auth.py 
# JWT payload에서 추출한 이메일로 사용자 조회
result = await db.execute(select(User).where(User.email == email))
user = result.scalar_one_or_none()
fastapi/app/services/auth_service.py

프로젝트 전체에서 원시 SQL 문자열(text())이나 f-string 쿼리를 사용하지 않으며, SQLAlchemy의 select(), where(), insert() 등 Expression Language만 사용합니다. 이는 SQL Injection을 구조적으로 차단하는 가장 효과적인 방법입니다.

Q9 CORS 정책 설정은? (configurable origins, credentials=True) +

FastAPI의 CORSMiddleware를 사용하여 교차 출처 요청을 관리합니다. 허용 출처(origins)는 환경변수로 설정 가능하며, 쿠키 전송을 위해 allow_credentials=True가 활성화되어 있습니다.

CORS 미들웨어 등록

fastapi/app/main.py 
# CORS middleware
app.add_middleware(
    CORSMiddleware,
    allow_origins=settings.cors_origins,    # 환경변수에서 파싱된 리스트
    allow_credentials=True,                 # 쿠키(HttpOnly) 전송 허용
    allow_methods=["*"],                    # 모든 HTTP 메서드 허용
    allow_headers=["*"],                    # 모든 헤더 허용 (X-CSRF-Token 포함)
)

CORS 출처 설정

fastapi/app/core/config.py 
# 환경변수 (쉼표 구분)
CORS_ALLOWED_ORIGINS: str = "http://localhost:3000,http://localhost:5173"

@property
def cors_origins(self) -> List[str]:
    """Parse CORS_ALLOWED_ORIGINS into a list."""
    return [origin.strip() for origin in self.CORS_ALLOWED_ORIGINS.split(",")]

환경별 설정

환경CORS_ALLOWED_ORIGINS
로컬 개발http://localhost:3000,http://localhost:5173
프로덕션https://demo.adelie-invest.com (.env에서 설정)

CORS + 쿠키 인증의 관계

  • allow_credentials=True이면 allow_origins에 "*"를 사용할 수 없음 (브라우저 보안 정책)
  • 따라서 반드시 명시적 출처 목록을 지정해야 함
  • HttpOnly 쿠키(accessToken, refreshToken)가 cross-origin 요청에서도 전송되려면 이 설정이 필수
  • allow_headers=["*"]로 X-CSRF-Token 커스텀 헤더도 허용

프로덕션에서는 nginx가 리버스 프록시 역할을 하므로 프론트엔드와 백엔드가 동일 도메인(demo.adelie-invest.com)에서 서빙되어, 실질적으로 CORS 이슈가 발생하지 않습니다. 로컬 개발 시에만 포트가 달라(3000 vs 8082) CORS 설정이 필요합니다.

Q10 에러 정보 누출 방지는? (generic 500 + error ID, traceback 미노출) +

처리되지 않은 예외가 발생하면 고유 error_id만 클라이언트에 반환하고, 실제 스택 트레이스는 서버 로그에만 기록합니다. 이를 통해 공격자가 내부 구조(파일 경로, 라이브러리 버전, DB 스키마 등)를 파악하는 것을 방지합니다.

글로벌 예외 핸들러

fastapi/app/main.py 
@app.exception_handler(Exception)
async def global_exception_handler(request: Request, exc: Exception):
    """처리되지 않은 예외를 안전한 JSON 응답으로 변환 (민감 정보 노출 방지)."""
    error_id = uuid.uuid4().hex[:12]       # 12자리 고유 ID 생성

    # 서버 로그에만 전체 정보 기록 (운영팀 디버깅용)
    logger.error(
        "Unhandled exception [error_id=%s] %s %s: %s\n%s",
        error_id,
        request.method,
        request.url.path,
        exc,
        traceback.format_exc(),            # 스택 트레이스는 로그에만!
    )

    # 클라이언트에는 일반적인 메시지 + error_id만 반환
    return JSONResponse(
        status_code=500,
        content={
            "status": "error",
            "message": "Internal server error",
            "error": {
                "code": "internal_error",
                "error_id": error_id,
                "detail": "An unexpected error occurred. "
                          "Please contact support with the error_id.",
            },
        },
    )

HTTP 예외 핸들러 (구조화된 응답)

@app.exception_handler(HTTPException)
async def http_exception_handler(request: Request, exc: HTTPException):
    """HTTP 예외를 전역 에러 포맷으로 변환."""
    detail = exc.detail
    if isinstance(detail, dict):
        message = detail.get("message") or detail.get("error") or "Request failed"
        error = {k: v for k, v in detail.items() if k != "message"}
    else:
        message = str(detail) if detail else "Request failed"
        error = None
    return JSONResponse(
        status_code=exc.status_code,
        content={"status": "error", "message": message, "error": error},
    )

Q11 DB IntegrityError 처리 패턴은? (rollback + 400 응답) +

SQLAlchemy의 IntegrityError(유니크 제약, 외래키 위반 등)를 catch하여 명시적 rollback + 사용자 친화적 400 응답을 반환합니다. DB 에러 메시지가 클라이언트에 직접 노출되지 않으며, 서비스별로 적절한 후처리 전략을 적용합니다.

패턴 1: 회원가입 (rollback + 에러 반환)

fastapi/app/services/auth_service.py 
user = User(
    email=email,
    username=username,
    password_hash=_pwd_context.hash(password),
    difficulty_level=normalized_level,
)
db.add(user)
try:
    await db.commit()
    await db.refresh(user)
except IntegrityError:
    await db.rollback()                              # 트랜잭션 롤백
    raise HTTPException(
        status_code=status.HTTP_400_BAD_REQUEST,
        detail="이미 등록된 사용자 정보입니다.",      # DB 에러 메시지 숨김
    )

사전에 SELECT로 중복을 확인하더라도, 동시 요청(race condition)으로 인한 중복 삽입을 DB 유니크 제약이 최종 방어합니다. 이 경우 IntegrityError가 발생하며, 안전하게 처리합니다.

패턴 2: 포트폴리오 (rollback + 기존 레코드 조회)

fastapi/app/services/portfolio_service.py 
portfolio = UserPortfolio(
    user_id=user_id,
    portfolio_name="내 포트폴리오",
    initial_cash=1_000_000,
    current_cash=1_000_000,
)
db.add(portfolio)
try:
    await db.flush()
except IntegrityError:
    await db.rollback()
    # 이미 존재하는 포트폴리오를 다시 조회 (idempotent 처리)
    result = await db.execute(stmt)
    portfolio = result.scalars().first()
else:
    await db.commit()
    await db.refresh(portfolio, attribute_names=["holdings", "trades"])

두 패턴의 차이

패턴사용처전략
rollback + 400 에러회원가입 (email/username unique)중복은 비즈니스 에러 → 클라이언트에 알림
rollback + 기존 조회포트폴리오 생성 (user_id unique)멱등성(idempotent) → 이미 있으면 그대로 반환

방어 계층

  1. 사전 SELECT 검증: 대부분의 중복을 잡아냄 (사용자 경험 개선)
  2. DB 유니크 제약: race condition 시 최종 방어선
  3. IntegrityError catch: 안전한 롤백 + 적절한 응답

모든 IntegrityError 처리에서 DB 원본 에러 메시지(테이블명, 컬럼명, 제약조건명 등)는 클라이언트에 노출하지 않고, 한글로 된 사용자 친화적 메시지만 반환합니다.

Q12 등록 시 입력 검증 체인은? (EmailStr → 도메인 차단 → 사용자명 검증 → 난이도 검증) +

회원가입 시 5단계 검증 체인이 순차적으로 실행됩니다. 각 단계를 통과하지 못하면 즉시 400 에러를 반환하여, 불필요한 DB 작업을 방지합니다.

검증 체인 전체 흐름

fastapi/app/services/auth_service.py — register_user() 
async def register_user(db, *, email, password, username, difficulty_level):
    settings = get_settings()

    # [1단계] 이메일 형식 + 도메인 차단
    email = _validate_email_domain(email, settings.registration_blocked_domains)

    # [2단계] 이메일 중복 확인
    existing = await db.execute(select(User).where(User.email == email))
    if existing.scalar_one_or_none():
        raise HTTPException(status_code=400, detail="이미 등록된 이메일입니다.")

    # [3단계] 사용자명 금칙어 검증
    if not username:
        username = email.split("@", 1)[0]
    pattern = re.compile(settings.REGISTRATION_BLOCKED_USERNAME_PATTERN, re.IGNORECASE)
    _validate_username(username, pattern)

    # [4단계] 사용자명 중복 확인
    existing_username = await db.execute(select(User).where(User.username == username))
    if existing_username.scalar_one_or_none():
        raise HTTPException(status_code=400, detail="이미 사용 중인 사용자명입니다.")

    # [5단계] 난이도 레벨 검증
    normalized_level = (difficulty_level or "beginner").strip().lower()
    allowed_levels = {"beginner", "elementary", "intermediate"}
    if normalized_level not in allowed_levels:
        raise HTTPException(status_code=400,
            detail="difficulty_level은 beginner, elementary, intermediate 중 하나여야 합니다.")

    # 모든 검증 통과 → DB 저장
    user = User(
        email=email,
        username=username,
        password_hash=_pwd_context.hash(password),
        difficulty_level=normalized_level,
    )
    db.add(user)
    try:
        await db.commit()
    except IntegrityError:
        await db.rollback()
        raise HTTPException(status_code=400, detail="이미 등록된 사용자 정보입니다.")

각 검증 세부 내용

  • EmailStr (Pydantic): RFC 5322 기반 이메일 형식 검증, 잘못된 형식은 422 반환
  • 도메인 차단: tempmail.com 등 5개 일회용 이메일 도메인 차단
  • 금칙어 패턴: .*(admin|test|root|관리자|운영자|욕설).* 대소문자 무시
  • 난이도 검증: 화이트리스트 방식 — beginner, elementary, intermediate만 허용
  • IntegrityError: 동시 요청 시 DB 유니크 제약으로 최종 방어

이 체인 구조는 빠른 실패(fail-fast) 원칙을 따릅니다. DB 쿼리가 필요 없는 형식/도메인 검증을 먼저 수행하고, 비용이 큰 DB 중복 확인은 나중에 실행하여 불필요한 리소스 소비를 최소화합니다.
