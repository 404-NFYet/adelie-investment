Q3 JWT 인증 + HttpOnly 쿠키 구현은? +

JWT 기반 인증에서 Access Token과 Refresh Token을 모두 HttpOnly 쿠키로 전송합니다. XSS 공격 시에도 JavaScript에서 토큰에 접근할 수 없으므로 보안성이 높습니다.

토큰 생성 (auth_service.py)

fastapi/app/services/auth_service.py (L48-62) 
def _build_token(subject: str, expires_in: int,
                 extra_claims: dict | None = None) -> str:
    settings = get_settings()
    now = datetime.now(timezone.utc)
    payload = {
        "sub": subject,       # 이메일 (사용자 식별자)
        "iat": int(now.timestamp()),
        "exp": int((now + timedelta(seconds=expires_in)).timestamp()),
    }
    if extra_claims:
        payload.update(extra_claims)  # refresh엔 jti, uid 추가
    return jwt.encode(
        payload,
        _get_jwt_key(settings.JWT_SECRET),
        algorithm=settings.JWT_ALGORITHM,  # HS256
    )

Q4 CSRF 방어 (Double-Submit Cookie) 구현은? +

HttpOnly 쿠키 인증 사용 시 CSRF 공격을 방어하기 위해 Double-Submit Cookie 패턴을 구현합니다. 서버가 랜덤 토큰을 csrfToken 쿠키(httponly=false)에 저장하고, 클라이언트가 이를 읽어 X-CSRF-Token 헤더로 전송해야 합니다.

CSRF 토큰 발급

fastapi/app/api/routes/auth.py (L91-103) 
def _set_csrf_cookie(response: Response) -> str:
    settings = get_settings()
    token = secrets.token_urlsafe(32)  # 256-bit 랜덤 토큰
    response.set_cookie(
        key=settings.AUTH_CSRF_COOKIE_NAME,  # "csrfToken"
        value=token,
        httponly=False,    # JS에서 읽을 수 있어야 함 (핵심!)
        secure=settings.AUTH_CSRF_COOKIE_SECURE,
        samesite=settings.AUTH_CSRF_COOKIE_SAMESITE,
        path=settings.AUTH_CSRF_COOKIE_PATH,
        domain=settings.AUTH_CSRF_COOKIE_DOMAIN,
    )
    return token

CSRF 미들웨어 검증

fastapi/app/main.py (L199-227) 
@app.middleware("http")
async def csrf_middleware(request: Request, call_next):
    settings = get_settings()
    # GET, HEAD, OPTIONS는 검증 생략 (안전 메서드)
    if request.method in {"GET", "HEAD", "OPTIONS"}:
        return await call_next(request)

    # 인증 관련 경로는 CSRF 검증 제외
    path = request.url.path
    csrf_exempt_prefixes = {
        "/api/v1/auth/login", "/api/v1/auth/register",
        "/api/v1/auth/refresh", "/api/v1/auth/csrf",
    }
    if path in csrf_exempt_prefixes:
        return await call_next(request)

    csrf_cookie = request.cookies.get(settings.AUTH_CSRF_COOKIE_NAME)
    if not csrf_cookie:
        # 쿠키 없음 = Authorization 헤더 방식 → CSRF 불필요
        return await call_next(request)

    # 쿠키 값과 헤더 값 비교 (Double-Submit)
    csrf_header = request.headers.get(settings.AUTH_CSRF_HEADER_NAME)
    if not csrf_header or csrf_cookie != csrf_header:
        return JSONResponse(
            status_code=403,
            content={"detail": "CSRF token missing or invalid"},
        )
    return await call_next(request)

보안 설계 포인트

  • Authorization 헤더 방식은 CSRF에 면역: CSRF 쿠키가 없으면 검증을 건너뛰어, Bearer 토큰 방식 클라이언트도 정상 동작
  • SameSite=Lax: 크로스사이트 POST에서 쿠키가 전송되지 않아 추가 방어층 제공
  • secrets.token_urlsafe(32): 256비트 암호학적 안전 랜덤 토큰 사용
  • 로그인/회원가입/리프레시 제외: 인증 전 상태에서 CSRF 토큰이 없으므로 exemption 처리
Q5 Rate Limiting (Redis Lua 슬라이딩 윈도우) 구현은? +

IP 기반 슬라이딩 윈도우 100 req/min 제한을 Redis의 Sorted Set + Lua 스크립트로 원자적으로 처리합니다. Redis 장애 시에는 rate limit을 적용하지 않고 요청을 통과시키는 graceful degradation 전략을 사용합니다.

Lua 스크립트 (원자적 연산)

fastapi/app/middleware/rate_limit.py (L15-33) 
SLIDING_WINDOW_LUA = """
local key = KEYS[1]
local now = tonumber(ARGV[1])       -- 현재 시각 (ms)
local window = tonumber(ARGV[2])    -- 윈도우 크기 (60000ms = 1분)
local limit = tonumber(ARGV[3])     -- 최대 요청 수 (100)

-- 1. 윈도우 밖의 오래된 요청 제거
redis.call('ZREMRANGEBYSCORE', key, 0, now - window)
-- 2. 현재 윈도우 내 요청 수 계산
local count = redis.call('ZCARD', key)

if count < limit then
    -- 허용: 현재 시각을 score/member로 추가
    redis.call('ZADD', key, now, now)
    redis.call('EXPIRE', key, math.ceil(window / 1000))
    return {1, limit - count - 1, 0}   -- {allowed, remaining, reset}
else
    -- 거부: 가장 오래된 요청의 만료 시각 계산
    local oldest = redis.call('ZRANGE', key, 0, 0, 'WITHSCORES')
    local reset = oldest[2] and (tonumber(oldest[2]) + window)
                  or (now + window)
    return {0, 0, reset}                -- {denied, 0, reset_ms}
end
"""

미들웨어 실행

fastapi/app/middleware/rate_limit.py (L42-77) 
class RateLimitMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        if request.url.path in _EXEMPT_PATHS:  # /metrics, /docs 등 제외
            return await call_next(request)

        client_ip = request.client.host if request.client else "unknown"
        try:
            cache = await get_redis_cache()
            if cache.client:
                now_ms = int(time.time() * 1000)
                k = key_rate_limit("ip", client_ip)  # "dev:rl:ip:1.2.3.4"
                result = await cache.client.eval(
                    SLIDING_WINDOW_LUA, 1, k,
                    now_ms, WINDOW_MS, RATE_LIMIT
                )
                allowed, remaining, reset_ms = result
                if not allowed:
                    retry_after = max(1, int((reset_ms - now_ms) / 1000))
                    return Response(
                        status_code=429,
                        headers={
                            "X-RateLimit-Limit": str(RATE_LIMIT),
                            "X-RateLimit-Remaining": "0",
                            "X-RateLimit-Reset": str(int(reset_ms / 1000)),
                            "Retry-After": str(retry_after),
                        },
                    )
        except Exception as e:
            # Redis 장애 시 rate limit 미적용 (서비스 가용성 우선)
            logger.warning(f"RateLimit Redis error (fallback=allow): {e}")

        return await call_next(request)

def _extract_db_operation(statement: str) -> str:
    """SQL 문에서 연산 유형 추출 (select/insert/update/delete)."""
    if not statement:
        return "other"
    op = statement.lstrip().split(None, 1)[0].lower()
    if op in {"select", "insert", "update", "delete"}:
        return op
    return "other"

@event.listens_for(engine.sync_engine, "after_cursor_execute")

@event.listens_for(engine.sync_engine, "handle_error")
def _handle_db_error(exception_context):
    statement = getattr(exception_context, "statement", "") or ""
    DB_QUERY_TOTAL.labels(_extract_db_operation(statement), "fail").inc()

Q7 Refresh Token 회전 및 재사용 탐지는? +

Refresh Token Rotation: 매 갱신 시 기존 토큰을 폐기하고 새 토큰을 발급합니다. Reuse Detection: 이미 사용된 토큰이 다시 제출되면 해당 사용자의 모든 Refresh Token을 즉시 폐기합니다(토큰 탈취 대응).

    client = await _require_redis()

    user_key = f"auth:rt:uid:{user_id}"    # 사용자별 jti 집합
    async with client.pipeline(transaction=True) as pipe:
        pipe.setex(token_key, ttl, token_hash)  # SET + EXPIRE 원자적
        pipe.sadd(user_key, jti)                 # 사용자의 jti 목록에 추가
        pipe.expire(user_key, ttl)
        await pipe.execute()

재사용 탐지 + 토큰 회전

fastapi/app/services/auth_service.py (L275-350) 
async def refresh_tokens(db: AsyncSession, *, refresh_token: str) -> dict:
    # 1. JWT 디코딩 → email, jti, uid 추출
    payload = jwt.decode(refresh_token, ...)
    jti = payload.get("jti")
    uid = payload.get("uid")

    # 2. Redis에서 저장된 해시 조회
    client = await _require_redis()
    token_key = f"auth:rt:{jti}"
    stored_hash = await client.get(token_key)
    incoming_hash = _hash_token(refresh_token)  # SHA-256

    # 3. 재사용 탐지: 해시 불일치 또는 미존재
    if not stored_hash or stored_hash != incoming_hash:
        # 모든 Refresh Token 즉시 폐기 (보안 조치)
        await _revoke_user_refresh_tokens(user.id)
        raise HTTPException(
            status_code=401,
            detail="Refresh token reuse detected",
        )

    # 4. 토큰 회전: 기존 삭제 + 새 토큰 생성 (원자적)
    new_refresh_jti = uuid.uuid4().hex
    new_refresh_token = _build_token(
        email, refresh_ttl,
        extra_claims={"jti": new_refresh_jti, "uid": user.id},
    )
    async with client.pipeline(transaction=True) as pipe:
        pipe.delete(token_key)              # 기존 토큰 삭제
        pipe.srem(user_key, jti)            # 기존 jti 제거
        pipe.setex(f"auth:rt:{new_refresh_jti}", refresh_ttl,
                   _hash_token(new_refresh_token))  # 새 토큰 저장
        pipe.sadd(user_key, new_refresh_jti)
        pipe.expire(user_key, refresh_ttl)
        await pipe.execute()

전체 폐기 (보안 비상)

fastapi/app/services/auth_service.py (L119-131) 
async def _revoke_user_refresh_tokens(user_id: int) -> None:
    """사용자의 모든 Refresh Token 폐기 (탈취 의심 시)."""
    client = await _require_redis()
    user_key = f"auth:rt:uid:{user_id}"
    jtis = await client.smembers(user_key)     # 모든 jti 조회
    if not jtis:
        await client.delete(user_key)
        return
    async with client.pipeline(transaction=True) as pipe:
        for jti in jtis:
            pipe.delete(f"auth:rt:{jti}")      # 개별 토큰 모두 삭제
        pipe.delete(user_key)                   # jti 집합 삭제
        await pipe.execute()

보안 흐름 요약

  1. 정상 갱신: 기존 RT 삭제 → 새 RT 발급 → Redis에 새 해시 저장 (1회성)
  2. 탈취된 RT 사용: 이미 삭제된 jti이므로 stored_hash == None → 재사용 감지
  3. 재사용 감지 시: 해당 사용자의 모든 RT를 Redis에서 삭제 → 공격자와 정상 사용자 모두 재로그인 필요
  4. 해시 저장: SHA-256으로 해시 후 저장하여 Redis 데이터 유출 시에도 원본 토큰 복원 불가

async def get_current_user(
    request: Request,
    credentials: HTTPAuthorizationCredentials = Depends(security),
    db: AsyncSession = Depends(get_db),
) -> dict:
    settings = get_settings()
    # Authorization 헤더 우선, 없으면 쿠키 fallback
    token = credentials.credentials if credentials else None
    if not token and request:
        token = request.cookies.get(settings.AUTH_ACCESS_COOKIE_NAME)
    if not token:
        raise HTTPException(status_code=401, detail="Authentication required")
    payload = _decode_token(token)
    return await _resolve_user_from_payload(payload, db)
    # 반환: {"id": int, "email": str, "username": str, "difficulty_level": str}

                    "level": record.levelname,

                    "message": record.getMessage(),

    response = await call_next(request)

    # 스트리밍/문서/비JSON 응답은 제외

        return response
    if "application/json" not in content_type:

        return response

    # 자동 래핑

        wrapped = {"status": "error", "message": "Request failed",
                    "error": payload}
    else:
        wrapped = {"status": "success", "data": payload}
