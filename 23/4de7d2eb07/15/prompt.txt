### 4-B: `utils/context_collector.py` (신규)

SSH를 통해 해당 서버의 실시간 상태를 수집. 30초 캐시.

```python
@st.cache_data(ttl=30)
def get_server_context(ssh_alias: str) -> dict:
    """git 브랜치/상태 + 실행 중인 Docker 컨테이너 목록 반환"""
    ssh = get_ssh_client(ssh_alias)  # utils/ssh.py 재사용
    if not ssh:
        return {"available": False}

    # git 상태
    _, git_branch, _ = ssh.exec_command(
        "cd ~/adelie-investment && git branch --show-current"
    )
    _, git_status, _ = ssh.exec_command(
        "cd ~/adelie-investment && git status --short | head -5"
    )
    _, git_log, _ = ssh.exec_command(
        "cd ~/adelie-investment && git log --oneline -3"
    )
    # Docker 컨테이너
    _, containers, _ = ssh.exec_command(
        "docker ps --format '{{.Names}}\t{{.Status}}' | head -10"
    )
    return {
        "available": True,
        "branch": git_branch.read().decode().strip(),
        "git_changes": git_status.read().decode().strip(),
        "recent_commits": git_log.read().decode().strip(),
        "containers": containers.read().decode().strip(),
    }
```

### 4-C: `utils/claude_client.py` (신규)

```python
import anthropic
from config import CLAUDE_API_KEY, CLAUDE_MODEL

def stream_response(messages: list[dict], system: str):
    """Claude API 스트리밍 — generator로 텍스트 청크 yield"""
    client = anthropic.Anthropic(api_key=CLAUDE_API_KEY)
    with client.messages.stream(
        model=CLAUDE_MODEL,
        max_tokens=1500,
        system=system,
        messages=messages,
    ) as stream:
        for text in stream.text_stream:
            yield text
```

def render():
    st.title("💬 서버 어시스턴트")

    identity_key = st.session_state.get("identity")

    # ── Step 1: 신원 선택 ──────────────────────
    if not identity_key:
        st.markdown("**나는 누구인가요?**  _(대화 기록은 이 세션 동안 유지됩니다)_")
        cols = st.columns(5)
        for i, (key, info) in enumerate(IDENTITIES.items()):
            if cols[i].button(f"{info['icon']} {info['name']}", use_container_width=True):
                st.session_state["identity"] = key
                st.session_state["identity_ctx_done"] = False
                st.rerun()
        return

    info = IDENTITIES[identity_key]

    # ── Step 2: 작업 컨텍스트 선택 ──────────────
    if not st.session_state.get("identity_ctx_done"):
        st.subheader(f"{info['icon']} {info['name']} 님 — 작업 컨텍스트 (선택 사항)")
        task = st.radio(
            "지금 하는 작업:", ["(선택 안 함)"] + _task_options(info["role"]),
            horizontal=True, key="task_radio",
        )
        c1, c2 = st.columns([1, 5])
        if c1.button("✅ 시작하기"):
            st.session_state["current_task"] = None if task == "(선택 안 함)" else task
            st.session_state["identity_ctx_done"] = True
            st.rerun()
        if c2.button("↩️ 다시 선택"):
            st.session_state.pop("identity", None)
            st.rerun()
        return

    # ── Step 3: 채팅 인터페이스 ─────────────────
    task = st.session_state.get("current_task")
    ctx  = get_server_context(identity_key)   # SSH, 30초 캐시

    # 헤더 (컴팩트)
    hcol1, hcol2 = st.columns([6, 1])
    hcol1.markdown(
        f"**{info['icon']} {info['name']}** | `{identity_key}` ({info['host']}) | `{ctx.get('branch', info['branch'])}`"
        + (f" | 📋 {task}" if task else "")
    )
    if hcol2.button("↔️ 전환"):
        for k in ("identity", "identity_ctx_done", "current_task"):
            st.session_state.pop(k, None)
        st.rerun()
    if ctx.get("containers"):
        st.caption(f"🐳 {ctx['containers'][:120]}")

    st.divider()

    # 시나리오 버튼 (역할 + 작업 기반)
    scenarios = _get_scenarios(info["role"], task)
    scols = st.columns(3)
    for i, (label, prompt) in enumerate(scenarios):
        if scols[i % 3].button(label, key=f"sc_{i}"):
            msgs = st.session_state.setdefault(f"msgs_{identity_key}", [])
            msgs.append({"role": "user", "content": prompt})
            st.rerun()

    st.divider()

    # 대화 이력
    for msg in st.session_state.get(f"msgs_{identity_key}", []):
        with st.chat_message(msg["role"]):
            st.markdown(msg["content"])

    # 입력창 + 스트리밍
    if user_input := st.chat_input("질문을 입력하세요...", key=f"inp_{identity_key}"):
        msgs = st.session_state.setdefault(f"msgs_{identity_key}", [])
        msgs.append({"role": "user", "content": user_input})
        with st.chat_message("assistant"):
            system = build_system_prompt(info, ctx, task)
            response = st.write_stream(stream_response(msgs[:-1] + [{"role":"user","content":user_input}], system))
        msgs.append({"role": "assistant", "content": response})
        st.rerun()

    # 초기화
    if st.button("🗑️ 대화 초기화"):
        st.session_state[f"msgs_{identity_key}"] = []
        st.rerun()
