```makefile
sync-dev-branches:
	@echo "prod-final → dev-final/* 브랜치 싱크..."
	@cd $(ROOT) && CURRENT=$$(git branch --show-current); \
	for BRANCH in dev-final/frontend dev-final/backend dev-final/chatbot dev-final/pipeline dev-final/infra; do \
		echo "  -> $$BRANCH"; \
		git checkout $$BRANCH 2>/dev/null || { echo "  브랜치 없음, 건너뜀: $$BRANCH"; continue; }; \
		git merge prod-final --no-edit 2>/dev/null || true; \
		git push origin $$BRANCH; \
	done; \
	git checkout $$CURRENT
	@echo "완료: 모든 dev-final/* 브랜치 싱크"
```
